  
async function getBreedList(){
  const response=await fetch('https://dog.ceo/api/breeds/list/all');
  const data=await response.json();
  createBreedList(data.message);
}




function createBreedList(data){
     document.getElementById("breed").innerHTML=
     `<select onchange='loadBreed(this.value)'>
     <option>Choose a Dog Breed</option>
     ${Object.keys(data).map(function(breed){
       return `<option>${breed}</option>`;
     }).join('')}
     </option></select>`
  }

async function loadBreed(breed){
  if(breed != 'Choose a Dog Breed'){
      
      const response=await fetch(`https://dog.ceo/api/breed/${breed}/images`)
      const data=await response.json();
      toggleImages(data.message);
  }
}

function toggleImages(images){

    document.getElementById('dogimg').src=images[Math.round(Math.random()*images.length)];
}
